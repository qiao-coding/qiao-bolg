## 高级配置示例

```jsx
import { useState, useCallback } from 'react';
import MarkdownEditor from './markdownEditor';

export default function AdvancedEditor() {
  const [content, setContent] = useState('');
  const [theme, setTheme] = useState('light');

  // 自定义保存逻辑
  const handleSave = useCallback(async (content) => {
    try {
      const response = await fetch('/api/save-content', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({ content }),
      });

      if (response.ok) {
        console.log('内容保存成功');
      } else {
        console.error('保存失败');
      }
    } catch (error) {
      console.error('保存出错:', error);
    }
  }, []);

  // 内容变化处理（带防抖）
  const handleContentChange = useCallback(((content) => {
    setContent(content);
    // 可以在这里实现自动保存逻辑
  }), []);

  return (
    <div className="advanced-editor">
      <div className="editor-controls">
        <button onClick={() => setTheme(theme === 'light' ? 'dark' : 'light')}>
          切换到 {theme === 'light' ? '暗色' : '亮色'} 主题
        </button>
      </div>
      
      <MarkdownEditor
        value={content}
        onChange={handleContentChange}
        onSave={handleSave}
        theme={theme}
        initialMode="split"
      />
    </div>
  );
}
```

## 自定义工具栏示例

如果需要自定义工具栏按钮，可以这样实现：

```jsx
import { useState } from 'react';
import MarkdownEditor from './markdownEditor';

// 自定义工具栏按钮
const CustomToolbar = ({ onInsert }) => {
  const handleInsertImage = () => {
    const imageUrl = prompt('请输入图片链接:');
    if (imageUrl) {
      onInsert(`![图片描述](${imageUrl})`);
    }
  };

  return (
    <div className="custom-toolbar">
      <button onClick={handleInsertImage} title="插入图片">
        🖼️
      </button>
    </div>
  );
};

export default function CustomEditor() {
  const [content, setContent] = useState('');

  const handleInsert = (text) => {
    // 插入文本到编辑器
    setContent(prev => prev + text);
  };

  return (
    <div>
      <CustomToolbar onInsert={handleInsert} />
      <MarkdownEditor
        value={content}
        onChange={setContent}
      />
    </div>
  );
}
```

## 与表单集成示例

```jsx
import { useState } from 'react';
import MarkdownEditor from './markdownEditor';

export default function BlogForm() {
  const [title, setTitle] = useState('');
  const [content, setContent] = useState('');
  const [tags, setTags] = useState('');

  const handleSubmit = async (e) => {
    e.preventDefault();
    
    const blogData = {
      title,
      content,
      tags: tags.split(',').map(tag => tag.trim()),
      createdAt: new Date().toISOString(),
    };

    try {
      const response = await fetch('/api/blog', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify(blogData),
      });

      if (response.ok) {
        alert('博客发布成功!');
        // 重置表单
        setTitle('');
        setContent('');
        setTags('');
      } else {
        alert('发布失败');
      }
    } catch (error) {
      console.error('发布出错:', error);
      alert('发布失败');
    }
  };

  return (
    <form onSubmit={handleSubmit} className="blog-form">
      <div className="form-group">
        <label htmlFor="title">标题</label>
        <input
          type="text"
          id="title"
          value={title}
          onChange={(e) => setTitle(e.target.value)}
          required
        />
      </div>

      <div className="form-group">
        <label htmlFor="content">内容</label>
        <MarkdownEditor
          value={content}
          onChange={setContent}
          initialMode="split"
          showStatusBar={true}
        />
      </div>

      <div className="form-group">
        <label htmlFor="tags">标签 (用逗号分隔)</label>
        <input
          type="text"
          id="tags"
          value={tags}
          onChange={(e) => setTags(e.target.value)}
          placeholder="javascript, react, nextjs"
        />
      </div>

      <button type="submit">发布博客</button>
    </form>
  );
}
```

## 响应式设计示例

```jsx
import { useState, useEffect } from 'react';
import MarkdownEditor from './markdownEditor';

export default function ResponsiveEditor() {
  const [content, setContent] = useState('');
  const [mode, setMode] = useState('split');
  const [isMobile, setIsMobile] = useState(false);

  useEffect(() => {
    const checkIsMobile = () => {
      setIsMobile(window.innerWidth < 768);
    };

    checkIsMobile();
    window.addEventListener('resize', checkIsMobile);

    return () => window.removeEventListener('resize', checkIsMobile);
  }, []);

  // 在移动端默认使用编辑模式
  useEffect(() => {
    if (isMobile) {
      setMode('edit');
    } else {
      setMode('split');
    }
  }, [isMobile]);

  return (
    <div className="responsive-editor">
      <h2>响应式 Markdown 编辑器</h2>
      <p>当前屏幕: {isMobile ? '移动端' : '桌面端'}</p>
      
      <MarkdownEditor
        value={content}
        onChange={setContent}
        initialMode={mode}
        showToolbar={!isMobile} // 移动端隐藏工具栏以节省空间
        showStatusBar={!isMobile}
      />
    </div>
  );
}
```

## 注意事项

1. **性能考虑**：对于非常大的文档，建议使用虚拟滚动或分页加载
2. **数据持久化**：建议实现本地存储以防止内容丢失
3. **安全性**：在服务端渲染 Markdown 内容时，需要进行 XSS 防护
4. **国际化**：编辑器支持多语言，需要配置相应的语言包

## 常见问题

### Q: 如何自定义渲染组件?
A: 可以通过 `components` 属性自定义特定 Markdown 元素的渲染方式。

### Q: 如何添加自定义语法?
A: 可以通过配置 remark/rehype 插件来支持自定义语法。

### Q: 编辑器卡顿怎么办?
A: 检查文档大小，对于大文档考虑使用虚拟滚动或分段渲染。

这个 Markdown 编辑器提供了丰富的功能和灵活的配置选项，可以根据具体需求进行定制。
